select lp.id,op.cname,lp.mobile,m.txt1,ab.name,lp.scene_pic,lp.utime
from labor_proj lp
left join out_proj op  on lp.proj_id = op.id
left join mtc_t m on m.id = lp.job and m.type = 'WTRADE'
left join addr_book ab on ab.id = lp.team_id 
where lp.labor_id = '{pk}'
